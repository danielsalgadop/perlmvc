- PÁGINA DE LOGIN:

# Creo el CGI empty
my $cgi = CGI->new(); 

 # Creo una session sin cookie
my $session = CGI::Session->new("driver:File", undef, {'Directory'=>'/tmp/encuestas/'}) or die CGI::Session->errstr;

# Cojo el ID de la session
my $sid = $session->id();

# Creo una cookie y la asocio a la session my $cookie = $cgi->cookie(-name=>"encuestas",
						  -value=> $session->id);

# Pinto la cabecera del cgi, con la cookie generada print $cgi->header(-cookie => $cookie );


- PÁGINA A LA QUE SALTAS O REDIRIGES:

# Vuelvo a crear un objeto CGI
my $cgi = CGI->new(); # Variable CGI.

# Recupero la cookie de mi cgi

# Recupero la session con la cookie recuperada
my $session = CGI::Session->new("driver:File", $cookie, {'Directory'=>'/tmp/encuestas/'}) or die CGI::Session->errstr; # Sesion que contiene la informacion del
